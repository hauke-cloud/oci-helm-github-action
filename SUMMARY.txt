   on:
     push:
       branches: [ main ]
   
   permissions:
     contents: write
     packages: write
   
   jobs:
     deploy:
       runs-on: ubuntu-latest
       steps:
         - uses: hauke-cloud/oci-helm-githubaction@v1
           with:
             image-name: ghcr.io/${{ github.repository }}
             oci-namespace: ${{ github.repository_owner }}/charts
             registry-password: ${{ secrets.GITHUB_TOKEN }}
